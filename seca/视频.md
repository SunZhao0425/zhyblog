    function createOrderid()
    {

        if($redis)

            $orderid = date("YmdHis") . ComHelper::getRandomNumber(999999,100000);
            $ret = $redis->get(RedisKey::getOrderid($orderid));

            }

            {
                //已经存在， 递归自己
                return self::createOrderid();
            }
        }
